----------------------- REST книг
# получить все книги
GET /books
# получить книгу по id
GET /books/{id}
# получить жанры книги
GET /books/{id}/genres
# получить комментарий по id книги
GET /books/{id}/comments
# получить жанры по id книги
GET /books/{id}/genres
# удалить книгу
DELETE /books/{id}
# добавить книгу
POST /books   
  payload - JSON с описанием книги
  возвращает сохраненную книгу
# изменить данные книги
PUT /books/{id}
  payload - JSON с описанием книги
  возвращает сохраненную книгу

----------------------- REST авторов
# получить всех авторов
GET /authors
# получить автора по id
GET /authors/{id}

# добавить автора
POST /authors
  payload - JSON с описанием автора
  возвращает сохраненного автора
# обновить автора
PUT /authors/{id}
  payload - JSON с описанием автора
  возвращает сохраненного автора  
# удалить автора
DELETE /authors/{id}
